int delay(int ms)
{

    for (int i = 0; i < 1000; i++)

    }

    return sum;
}

bool isExitCommand(const char* buf) {
    return strcmp(buf, "exit") == 0;

bool goStartPoint(int step, int answer) {
    return answer == 0 && step == QuestionType::RunOrTest;
